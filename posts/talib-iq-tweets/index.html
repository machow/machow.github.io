library(mirt)
## Loading required package: stats4

library(mirtCAT)
## Loading required package: shiny
N_ITEMS <- 30
AVG_EASE <- 2.5
params_low <- data.frame(
  a1 = 1,                          # discrimination (factor loadings)
  d = rnorm(N_ITEMS, AVG_EASE)     # ease (negative of difficulty)
  )

mod_low <- generate.mirt_object(params_low, '2PL')

plot(mod_low, type = "info", xlim = c(-3, 3))

library(mirt)

# Same model, but half the items are difficult, and half are easy
params_tail <- data.frame(
  a1 = 1,
  d = c(
    rnorm(N_ITEMS/2, AVG_EASE),
    rnorm(N_ITEMS/2, -AVG_EASE)
    )
  )

mod_tail <- generate.mirt_object(params_tail, '2PL')

plot(mod_tail, type = "info", xlim = c(-3, 3))
